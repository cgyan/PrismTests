cmake_minimum_required(VERSION 3.10.2 FATAL_ERROR)

project(PrismTests)

option(integ "BuildIntegrationTests" OFF)
option(unit "BuildUnitTests" ON)

if (integ AND unit)
    message("PrismTests: build integration tests = YES")
    message("PrismTests: build unit tests = YES")
    add_subdirectory(integration)
elseif(integ)
    message("PrismTests: build integration tests = YES")
    message("PrismTests: build unit tests = NO")
    add_subdirectory(unit)
elseif(unit)
    message("PrismTests: build integration tests = NO")
    message("PrismTests: build unit tests = YES")
    add_subdirectory(unit)
else()
    message("PrismTests: build integration tests = NO")
    message("PrismTests: build unit tests = NO")
endif()
